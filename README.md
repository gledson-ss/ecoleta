    - npm init -y
    - npm install typescript -D
    - npm install express
    - npm install @types/express -D
    - npm install ts-node -D
    - npx tsc --init
    - npm install ts-node-dev -D

    - npm install sqlite3

    - npm run dev || yarn dev
